## FIX 1 — Hero image (LCP fix — biggest single win)

The hero section has a large full-width office/background image. This is the LCP element — the browser is waiting to download this entire image before showing anything.

### 1A. Find the hero image
Search for where the hero background image is defined:
```bash
grep -r "hero\|background-image\|bg-image\|office\|workspace" src --include="*.tsx" --include="*.css" -l
```

### 1B. If it's a CSS background-image — replace with Next.js Image component

**Before (slow):**
```css
.hero {
  background-image: url('/hero-bg.jpg');
  background-size: cover;
}
```

**After (fast):**
```tsx
import Image from "next/image";

// Inside the hero section JSX:
<div style={{ position: "relative", overflow: "hidden" }}>
  <Image
    src="/hero-bg.jpg"
    alt=""
    fill
    priority                    // ← CRITICAL: preloads immediately
    quality={75}                // ← reduce from default 100
    sizes="100vw"
    style={{ objectFit: "cover", objectPosition: "center" }}
  />
  <div style={{ position: "relative", zIndex: 1 }}>
    {/* hero text content goes here */}
  </div>
</div>
```

### 1C. Compress the hero image before uploading
The hero image must be under 200KB. If it's currently a JPG/PNG:
1. Go to https://squoosh.app
2. Upload the hero image
3. Set format to WebP, quality 75
4. Download and replace the file in `public/`

### 1D. Add preload hint in layout.tsx
```tsx
export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <head>
        <link
          rel="preload"
          as="image"
          href="/hero-bg.webp"
          type="image/webp"
        />
      </head>
      <body>{children}</body>
    </html>
  );
}
```

## FIX 2 — Remove render-blocking resources (FCP fix)

### 2A. Remove Google Fonts @import from globals.css
Find and DELETE this line from `src/app/globals.css`:
```css
/* DELETE THIS ENTIRE LINE: */
@import url('https://fonts.googleapis.com/css2?family=Inter...');
```

### 2B. Use next/font instead in layout.tsx
```tsx
import { Inter, Newsreader } from "next/font/google";

const inter = Inter({
  subsets: ["latin"],
  variable: "--font-inter",
  display: "swap",
  preload: true,
});

const newsreader = Newsreader({
  subsets: ["latin"],
  variable: "--font-newsreader",
  style: ["normal", "italic"],
  weight: ["400", "500"],
  display: "swap",
  preload: true,
});

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en" className={`${inter.variable} ${newsreader.variable}`}>
      <body>{children}</body>
    </html>
  );
}
```

### 2C. Update CSS font-family references
In `globals.css`, replace hardcoded font names:
```css
/* Before: */
body { font-family: 'Inter', sans-serif; }
h1, h2 { font-family: 'Newsreader', serif; }

/* After: */
body { font-family: var(--font-inter), sans-serif; }
h1, h2 { font-family: var(--font-newsreader), serif; }
```

## FIX 3 — Reduce Total Blocking Time (TBT fix)

TBT is red because heavy JavaScript is running on the main thread during page load.

### 3A. Make all platform pages static
Add to the top of EVERY `page.tsx` file (Home, FlutterFlow, Podio, Lovable, Replit, Retool, About):
```tsx
export const dynamic = "force-static";
export const revalidate = 3600;
```

### 3B. Dynamic import for heavy below-fold components
In every `page.tsx`, replace direct imports with dynamic imports:

```tsx
import dynamic from "next/dynamic";

// Replace these static imports:
// import FaqSection from "@/components/FaqSection";
// import ContactForm from "@/components/ContactForm";
// import IndustriesSection from "@/components/IndustriesSection";

// With dynamic imports (loads only when needed):
const FaqSection = dynamic(() => import("@/components/FaqSection"), {
  ssr: false,
  loading: () => <div style={{ minHeight: 300 }} aria-hidden="true" />,
});

const ContactForm = dynamic(() => import("@/components/ContactForm"), {
  ssr: false,
  loading: () => <div style={{ minHeight: 400 }} aria-hidden="true" />,
});

const IndustriesSection = dynamic(() => import("@/components/IndustriesSection"), {
  ssr: false,
  loading: () => <div style={{ minHeight: 300 }} aria-hidden="true" />,
});
```

### 3C. Defer non-critical animations
In `globals.css`, add animation-delay to everything that animates:
```css
/* Animations should not run during first paint */
* {
  animation-delay: 0.5s !important;
}

/* Remove transitions from above-fold elements */
nav, nav a, nav button {
  transition: none !important;
}
```

## FIX 5 — next.config.ts optimizations

Replace the entire `next.config.ts` with:

```ts
import type { NextConfig } from "next";

const nextConfig: NextConfig = {
  reactStrictMode: true,

  // Image optimization
  images: {
    formats: ["image/avif", "image/webp"],
    minimumCacheTTL: 86400,
    deviceSizes: [640, 750, 828, 1080, 1200],
    imageSizes: [16, 32, 48, 64, 96, 128, 256],
  },

  // Compress responses
  compress: true,

  // Remove powered-by header
  poweredByHeader: false,

  // Cache headers
  async headers() {
    return [
      {
        source: "/_next/static/(.*)",
        headers: [
          { key: "Cache-Control", value: "public, max-age=31536000, immutable" },
        ],
      },
      {
        source: "/(.*\\.webp|.*\\.avif|.*\\.png|.*\\.jpg|.*\\.svg|.*\\.ico)",
        headers: [
          { key: "Cache-Control", value: "public, max-age=86400, stale-while-revalidate=604800" },
        ],
      },
    ];
  },

  // Experimental: faster builds
  experimental: {
    optimizeCss: true,
    optimizePackageImports: ["lucide-react", "framer-motion"],
  },
};

export default nextConfig;
```

## FIX 6 — Add skeleton loading for above-fold content

So the page never shows blank white space during load:

Create `src/components/HeroSkeleton.tsx`:
```tsx
export default function HeroSkeleton() {
  return (
    <div
      style={{
        width: "100%",
        height: "100vh",
        background: "linear-gradient(90deg, #12161F 25%, #161B26 50%, #12161F 75%)",
        backgroundSize: "200% 100%",
        animation: "shimmer 1.5s infinite",
      }}
      aria-hidden="true"
    />
  );
}
```

Add to `globals.css`:
```css
@keyframes shimmer {
  0% { background-position: 200% 0; }
  100% { background-position: -200% 0; }
}
```
